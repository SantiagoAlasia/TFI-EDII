  ; Configurar ADC
    BANKSEL ADCON1
    MOVLW B'10000000'  ; Vref+ = VDD, resultado justificado a la izquierda
    MOVWF ADCON1

    BANKSEL ADCON0
    MOVLW B'00000001'  ; Canal AN0 (0000), ADC ON
    MOVWF ADCON0
    
    LECTURA_ADC:
    BSF ADCON0, GO      ; Iniciar conversión
ESPERAR_ADC:
    BTFSC ADCON0, GO    ; Esperar hasta que GO se limpie
    GOTO ESPERAR_ADC
